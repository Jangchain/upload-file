function minCut(s) {
  const n = s.length;
  const g = Array.from({ length: n }, () => Array(n).fill(true));
  const f = new Array(n).fill(0);

  for (let i = n - 1; i >= 0; --i) {
    for (let j = i + 1; j < n; ++j) {
      g[i][j] = s[i] === s[j] && g[i + 1][j - 1];
    }
  }

  for (let i = 0; i < n; ++i) {
    if (g[0][i]) {
      f[i] = 0;
    } else {
      f[i] = Number.MAX_SAFE_INTEGER;
      for (let j = 0; j < i; ++j) {
        if (g[j + 1][i]) {
          f[i] = Math.min(f[i], f[j] + 1);
        }
      }
    }
  }

  return f[n - 1];
}
